cmake_minimum_required(VERSION 3.8)

project(ModelViewer)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOUIC_SEARCH_PATHS ${CMAKE_SOURCE_DIR}/resource/ui)

# 屏蔽 MSVC 警告（C4819: 编码字符  C4244: 类型转换截断）
add_compile_options(
    "$<$<C_COMPILER_ID:MSVC>:/wd4819>"
    "$<$<C_COMPILER_ID:MSVC>:/wd4244>"
    "$<$<CXX_COMPILER_ID:MSVC>:/wd4819>"
    "$<$<CXX_COMPILER_ID:MSVC>:/wd4244>"
)

# Eigen 对齐配置（PCL 依赖）
add_definitions(-DEIGEN_MAX_ALIGN_BYTES=32)

set(EXECUTABLE_OUTPUT_PATH ${PROJECT_NAME}/bin)

# Qt 自动查找，如未找到请通过命令行指定路径：
#   cmake -DCMAKE_PREFIX_PATH=D:/Qt/Qt5.14.2/5.14.2/msvc2017_64 ..
find_package(Qt5 COMPONENTS Widgets REQUIRED)

include_directories(${CMAKE_SOURCE_DIR}/include)
include_directories(${CMAKE_SOURCE_DIR}/include/layer1/wrapper)
include_directories(${CMAKE_SOURCE_DIR}/include/layer2)
include_directories(${CMAKE_SOURCE_DIR}/include/layer3)
include_directories(${CMAKE_SOURCE_DIR}/thirdParty/glew/include)
include_directories(${CMAKE_SOURCE_DIR}/thirdParty/opencascade/include)
include_directories(${CMAKE_SOURCE_DIR}/thirdParty/glm/include)
include_directories(${CMAKE_SOURCE_DIR}/thirdParty/imgui/include)

# PCL 1.15.1 — 点云 IO 模块
set(PCL_ROOT ${CMAKE_SOURCE_DIR}/thirdParty/pcl)
include_directories(${PCL_ROOT}/include/pcl-1.15)
include_directories(${PCL_ROOT}/3rdParty/Eigen3/include/eigen3)
include_directories(${PCL_ROOT}/3rdParty/Boost/include/boost-1_87)
include_directories(${PCL_ROOT}/3rdParty/FLANN/include)
include_directories(${PCL_ROOT}/3rdParty/Qhull/include)

# GLEW 库路径（动态链接）
set(GLEW_LIB ${CMAKE_SOURCE_DIR}/thirdParty/glew/lib/x64/glew32.lib)
set(GLEW_DLL ${CMAKE_SOURCE_DIR}/thirdParty/glew/bin/x64/glew32.dll)

# OpenCASCADE 7.6 — STEP 解析所需模块
set(OCC_LIB_DIR ${CMAKE_SOURCE_DIR}/thirdParty/opencascade/lib)
set(OCC_DLL_DIR ${CMAKE_SOURCE_DIR}/thirdParty/opencascade/bin)
set(OCC_LIBS
    ${OCC_LIB_DIR}/TKernel.lib
    ${OCC_LIB_DIR}/TKMath.lib
    ${OCC_LIB_DIR}/TKG2d.lib
    ${OCC_LIB_DIR}/TKG3d.lib
    ${OCC_LIB_DIR}/TKGeomBase.lib
    ${OCC_LIB_DIR}/TKGeomAlgo.lib
    ${OCC_LIB_DIR}/TKBRep.lib
    ${OCC_LIB_DIR}/TKTopAlgo.lib
    ${OCC_LIB_DIR}/TKSTEP.lib
    ${OCC_LIB_DIR}/TKSTEPBase.lib
    ${OCC_LIB_DIR}/TKSTEP209.lib
    ${OCC_LIB_DIR}/TKSTEPAttr.lib
    ${OCC_LIB_DIR}/TKXSBase.lib
)

# PCL 1.15.1 — 点云 IO 所需 .lib
set(PCL_LIB_DIR ${PCL_ROOT}/lib)
set(PCL_LIBS
    ${PCL_LIB_DIR}/pcl_io.lib
    ${PCL_LIB_DIR}/pcl_io_ply.lib
    ${PCL_LIB_DIR}/pcl_common.lib
    ${PCL_LIB_DIR}/pcl_octree.lib
)

# PCL 第三方 .lib（Boost / FLANN / Qhull）
set(PCL_3RDPARTY_LIB_DIR ${PCL_ROOT}/3rdParty)
set(PCL_3RDPARTY_LIBS
    ${PCL_3RDPARTY_LIB_DIR}/Boost/lib/libboost_filesystem-vc143-mt-x64-1_87.lib
    ${PCL_3RDPARTY_LIB_DIR}/Boost/lib/libboost_iostreams-vc143-mt-x64-1_87.lib
    ${PCL_3RDPARTY_LIB_DIR}/Boost/lib/libboost_system-vc143-mt-x64-1_87.lib
    ${PCL_3RDPARTY_LIB_DIR}/FLANN/lib/flann_cpp.lib
    ${PCL_3RDPARTY_LIB_DIR}/FLANN/lib/flann.lib
    ${PCL_3RDPARTY_LIB_DIR}/Qhull/lib/qhullstatic.lib
    ${PCL_3RDPARTY_LIB_DIR}/Qhull/lib/qhull_r.lib
)

file(GLOB_RECURSE SOURCES
    "src/*.cpp"
    "thirdParty/imgui/src/*.cpp"
)

file(GLOB_RECURSE HEADERS
    "include/*.h"
)

file(GLOB UI_FILES
    "resource/ui/*.ui"
)

add_executable(${PROJECT_NAME} 
    WIN32
    ${SOURCES}
    ${HEADERS}
    ${UI_FILES}
)

target_link_libraries(${PROJECT_NAME}  Qt5::Widgets ${GLEW_LIB} opengl32 ${OCC_LIBS} ${PCL_LIBS} ${PCL_3RDPARTY_LIBS})

# 构建后将 DLL 拷贝到可执行文件目录
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    ${GLEW_DLL} $<TARGET_FILE_DIR:${PROJECT_NAME}>
)

# 拷贝 PCL 运行时 DLL
set(PCL_DLL_DIR ${PCL_ROOT}/bin)
file(GLOB PCL_DLLS
    "${PCL_DLL_DIR}/pcl_io.dll"
    "${PCL_DLL_DIR}/pcl_io_ply.dll"
    "${PCL_DLL_DIR}/pcl_common.dll"
    "${PCL_DLL_DIR}/pcl_octree.dll"
)
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    ${PCL_DLLS} $<TARGET_FILE_DIR:${PROJECT_NAME}>
)
file(GLOB OCC_DLLS "${OCC_DLL_DIR}/*.dll")
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    ${OCC_DLLS} $<TARGET_FILE_DIR:${PROJECT_NAME}>
)
